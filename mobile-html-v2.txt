function submitMobileFormV2(e) {
  e.preventDefault();
  var status = document.getElementById('mob-form-status-v2');
  status.textContent = 'Отправка...';
  status.className = 'mob-form-status';
  var form = e.target;
  var data = new FormData(form);
  fetch('https://formspree.io/f/xrbpglen', {
    method: 'POST',
    body: data,
    headers: { 'Accept': 'application/json' }
  })
  .then(function(r) {
    if (r.ok) {
      status.textContent = '✓ Заявка отправлена! Мы свяжемся с вами в ближайшее время.';
      status.className = 'mob-form-status success';
      form.reset();
    } else {
      return r.json().then(function(d) { throw d; });
    }
  })
  .catch(function() {
    status.textContent = 'Ошибка отправки. Пожалуйста, напишите нам в Telegram: @mit_assistent';
    status.className = 'mob-form-status error';
  });
}
</script>
